CREATE TABLE CLIENTE(
  TELEFONO TEXT PRIMARY KEY NOT NULL,
  NOMBRE TEXT NOT NULL,
  DIRECCION TEXT NOT NULL,
  CREATION_DATE  TIMESTAMP NOT NULL,
  UPDATE_DATE TIMESTAMP NOT NULL
 );

CREATE TABLE company
(
   id    integer   NOT NULL,
   name  VARCHAR (30) NOT NULL,
   cod   text      NOT NULL,
  CREATION_DATE  TIMESTAMP NOT NULL,
  UPDATE_DATE TIMESTAMP NOT NULL
);

CREATE TABLE telemarketer
(
   id_telemarketer  integer   NOT NULL,
   id_company       integer,
   name             text      NOT NULL,
   tel              text,
  CREATION_DATE  TIMESTAMP NOT NULL,
  UPDATE_DATE TIMESTAMP NOT NULL
);

CREATE TABLE vendedor
(
   id_vendedor  integer   NOT NULL,
   id_company   integer,
   name         text      NOT NULL,
   nivel        text      NOT NULL,
   tel          text,
  CREATION_DATE  TIMESTAMP NOT NULL,
  UPDATE_DATE TIMESTAMP NOT NULL
);

CREATE TABLE CITA(
   ID_COMPANY     INT REFERENCES COMPANY(ID),
   ID_VENDEDOR    INT REFERENCES VENDEDOR(ID_VENDEDOR),
   ID_TELEMARKETER     INT REFERENCES TELEMARKETER(ID_TELEMARKETER),
   CLIENTE_TEL  TEXT REFERENCES CLIENTE(TELEFONO),
   DETALLE           TEXT  NOT NULL,
   HORA          TIME  NOT NULL,
   FECHA            DATE NOT NULL,
  CREATION_DATE  TIMESTAMP NOT NULL,
  UPDATE_DATE TIMESTAMP NOT NULL,
   PRIMARY KEY (ID_VENDEDOR, HORA, FECHA)
   );
